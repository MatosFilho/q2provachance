package source;

public class ContaCorrente {
	private String nome;
	private String senha;
	private double saldo;
	private int id;
	
	ContaCorrente(Cliente cliente, String senha, int id) {
		this.nome = cliente.getNome();
		this.saldo = cliente.getDinheiro();
		this.senha = senha;
		this.id = id;
	}
	
}
